        let mqttClient = null;

        let currentRoom = "";

        let myName = "";

        let players = new Set();

        function switchView(viewId) {

            document.querySelectorAll('.view').forEach(v => v.classList.remove('active-view'));

            document.getElementById(viewId).classList.add('active-view');

        }

        function connectToGame() {

            myName = document.getElementById("username").value.trim();

            currentRoom = document.getElementById("roomcode").value.trim();

            if (!myName || !currentRoom) {

                alert("Bitte Name und Raum-Code eingeben!");

                return;

            }

            document.getElementById("status").style.display = "block";

            document.getElementById("status").innerText = "Verbinde zum Server...";

            // Verbindung zum kostenlosen, öffentlichen MQTT-Broker via WebSockets herstellen

            mqttClient = mqtt.connect('wss://broker.hivemq.com:8884/mqtt');

            mqttClient.on('connect', () => {

                document.getElementById("status").innerText = "Echtzeit-Modus aktiv!";

               

                // Raum-Kanal abonnieren

                const topic = `qrpong/room/${currentRoom}`;

                mqttClient.subscribe(topic);

                // Den anderen mitteilen, dass ich da bin

                sendNetworkMessage({ type: "JOIN", name: myName });

                document.getElementById("lobby-title").innerText = `RAUM: ${currentRoom}`;

                switchView("view-lobby");

                updatePlayerList();

            });

            mqttClient.on('message', (topic, message) => {

                const data = JSON.parse(message.toString());

                handleNetworkMessage(data);

            });

        }

        function sendNetworkMessage(obj) {

            if (mqttClient) {

                const topic = `qrpong/room/${currentRoom}`;

                mqttClient.publish(topic, JSON.stringify(obj));

            }

        }

        function handleNetworkMessage(data) {

            if (data.type === "JOIN") {

                players.add(data.name);

                updatePlayerList();

                // Dem neuen Spieler antworten, damit er weiß, dass ich auch im Raum bin

                sendNetworkMessage({ type: "ALIVE", name: myName });

            }

            else if (data.type === "ALIVE") {

                players.add(data.name);

                updatePlayerList();

            }

            else if (data.type === "SHOW_QUESTION") {

                document.getElementById("card-category").innerText = data.category.toUpperCase();

                document.getElementById("question-display").innerText = data.text;

                document.getElementById("card-trigger").innerText = `Ausgewählt von: ${data.by}`;

                switchView("view-card");

            }

            else if (data.type === "GO_LOBBY") {

                switchView("view-lobby");

            }

        }

        function updatePlayerList() {

            const listDiv = document.getElementById("player-list");

            listDiv.innerHTML = "";

            // Mich selbst hinzufügen

            let html = `<span class="player-badge">${myName} (Du)</span>`;

            // Die anderen hinzufügen

            players.forEach(p => {

                if (p !== myName) {

                    html += `<span class="player-badge">${p}</span>`;

                }

            });

            listDiv.innerHTML = html;

        }

        function selectCategory(category) {

            const pool = questionPool[category];

            // Zufällige Frage auswählen

            const randomText = pool[Math.floor(Math.random() * pool.length)];

           

            // Befehl an ALLE Handys im Raum senden, diese Frage anzuzeigen

            sendNetworkMessage({

                type: "SHOW_QUESTION",

                category: category,

                text: randomText,

                by: myName

            });

        }

        function sendBackToLobby() {

            // Befehl an ALLE Handys senden, zurück zur Auswahl zu gehen

            sendNetworkMessage({ type: "GO_LOBBY" });

        }
